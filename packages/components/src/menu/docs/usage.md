## Usage

```js
import '@olympus/components/menu'
```

```html
<ods-menu>
  <ods-menu-item>Label</ods-menu-item>
</ods-menu>
```

> React

```jsx
import { OdsMenu, OdsMenuItem } from '@olympus/components/react/menu'
const App = () => (
  <OdsMenu>
    <OdsMenuItem>Label</OdsMenuItem>
  </OdsMenu>
)
```
